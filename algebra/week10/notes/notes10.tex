  \section{Section 3.2---Products and Quotients of Monomials} % (fold)

  \subsection{Properties} % (fold)
  
  \begin{align*}
    b^n b^m & = b^{n + m} \\
    x^2 x^3 & = xx \cdot xxx \\
            & = xxxxx \\
            & = x^5 \\
    \\
    \del{b^n}^m & = b^{nm} \\
    \del{x^2}^3 & = x^2 \cdot x^2 \cdot x^3 \\
                & = x^6 \\
    \\
    \del{ab}^n & = a^n b^n \\
    \del{xy}^3 & = xy \cdot xy \cdot xy \\
               & = xxx \cdot yyy \\
               & = x^3 y^3 \\
    \\
    \frac{b^n}{b^m} & = b^{n - m} \\
    \frac{x^5}{x^2} & = \frac{xxxxx}{xx} \\
                    & = x^3 \\
  \end{align*}
