        \begin{parts}
          \part 
            \begin{align*}
              y &= 157.68 - 2.993 x \\

          \part From Figure 5.11, $r^2 = 63.1 \%$
            
          \part Since the correlation coefficient doesn't depend on the order,

          correlation coefficient is negative:

            r = - \sqrt{67.88} = -8.24
          \]
