  \subsection{Pitching} % (fold)
  
  \begin{align*}
    ERA  & = \frac{RUNS}{IP/9} \\
    WHIP & = \frac{BB + H}{IP} \\
  \end{align*}

  \subsection{Batting} % (fold)
  \begin{align*}
    SLG &= \frac{1B + (2 \cdot 2B) + (3 \cdot 3B) + (4 \cdot HR)}{AB} \\
    OBP &= \frac{H + BB + HBP}{AB + BB + SF + HBP} \\
    OPS &= OBP + SLG \\
  \end{align*}
